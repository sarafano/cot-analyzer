const sheetURL = "https://docs.google.com/spreadsheets/d/1OQoOsv37W-A9oU40sajwPm-Wnu511rYKcDPchGLhxnM/edit?usp=drivesdk"; async function fetchData() { const response = await fetch(sheetURL); const text = await response.text(); const rows = text.split('\n').map(row => row.split(',')); // Filtramos para pegar apenas as últimas 10 semanas para não sobrecarregar o gráfico const dataHistory = rows.slice(-10); const cotData = { EUR: [], GBP: [], CAD: [], GOLD: [] }; dataHistory.forEach(cols => { if (cols.length > 10) { const date = cols[0]; // Coluna A cotData.EUR.push({ date: date, net: parseInt(cols[1]) || 0 }); // Coluna B cotData.GBP.push({ date: date, net: parseInt(cols[4]) || 0 }); // Coluna E cotData.CAD.push({ date: date, net: parseInt(cols[6]) || 0 }); // Coluna G cotData.GOLD.push({ date: date, net: parseInt(cols[10]) || 0 }); // Coluna K } }); return cotData; }
